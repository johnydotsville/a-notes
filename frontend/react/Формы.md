## Предотвращение перезагрузки страницы при отправке формы

```react
function App() {
  const [posts, setPosts] = useState([
    {id: 1, title: 'Javascript', body: 'Javascript - это язык программирования.'},
    {id: 2, title: 'Java', body: 'Java - это язык программирования, в основном используется на бэкэнде.'},
    {id: 3, title: 'C#', body: 'C# - это как Java, только лучше.'}
  ]);

  const [postTitle, setPostTitle] = useState("");
  const [postContent, setPostContent] = useState("");

  const addNewPost = (event) => {
    event.preventDefault();  // <-- Предотвращение перезагрузки страницы при отправке формы
    // setPosts({id: 1, title: postTitle, body: postContent});
    console.log(postTitle);
    console.log(postContent);
  };

  return (
    <div className="App">
      <form>
        <MyInput type="text" placeholder="Заголовок поста" value={postTitle} onChange={event => setPostTitle(event.target.value)} />
        <MyInput type="text" placeholder="Содержимое поста" value={postContent} onChange={event => setPostContent(event.target.value)} />
        <MyButton onClick={addNewPost}>Создать</MyButton>
      </form>
      <PostList title={"Посты"} posts={posts}/>
    </div>
  );
}

export default App;
```
