```java
public interface Queue<E>
	extends Collection<E> {
        +.element();    +.peek();
        +.offer();      +.poll();
}
```

```java
public interface Deque<E>
	extends Queue<E> {
        +.pop();                     +.getFirst();
        +.push();                    +.getLast();
        +.addFirst();                +.offerFirst();
        +.addLast();                 +.offerLast();
        +.removeFirst();             +.peekFirst();
        +.removeLast();              +.peekLast();
        +.removeFirstOccurence();    +.pollFirst();
        +.removeLastOccurence()      +.pollLast();
        +.descendingIterator();
}
```
